Examples

Simple Get

require ("PhCURL.php");
$curl = new PhCURL();

//Some useful settings:
$curl->setConnectTimeout(2)
     ->loadCommonSettings();

$curl->setUrl("http://www.google.com")
     ->GET();

$htmlCode = $curl->getData();
$curl->close();
